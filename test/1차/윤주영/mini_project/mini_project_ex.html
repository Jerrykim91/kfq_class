df = []
for n in range( len(target_url_list) ):
    #  url    
    driver.get( target_url_list[n] )    
    
    # end   
    for n in range(2):
        driver.find_element_by_tag_name('body').send_keys(Keys.END)
        time.sleep(2)
   
    #   
    m = driver.find_elements_by_css_selector('#content>div>div>div>ul>li>div>div>span')
    path_length = len(m)
    
    #   
    date = driver.find_element_by_css_selector('div.btn_trip.btn_trip_departure.unset > a').text
    
    #  1 . YYYY.mm.dd / 
    p        = re.compile('(\d+).(\d+).(\d+).(\w)')

    # '2019.09.xx.'
    real_date = p.sub( '\g<1>-\g<2>-\g<3>', date ) 
    real_day =p.sub('\g<4>',date)
    
    airport_tickets = list()

    for num in range(1,path_length+1) :
        #   
        for doc in driver.find_elements_by_css_selector(' div.trip_result.ng-scope > ul > li:nth-child({})'.format(num)):
            #doc.text ==>'\n\n06:50\n07:55\n01 05\n37,500\n'
            #   
            airport_tickets.append(doc.text)
            
            
    #  2.      
    p        = re.compile('\d+.\d+')
    
    
    for ticket in airport_tickets:
        #  
        tmp = ticket.split('\n')
        
        #     dic  
        airport_dic = {
            'date' : real_date,
            'day'  : real_day,
            'name' : tmp[0],
            'start_time' : tmp[3],
            'arrival_time': tmp[6],
            'lead_time' : tmp[8],
            'seat' : tmp[9],
            'price' : int( p.findall(tmp[10])[0].replace(',','') ),
            'age' : tmp[11]
        }
        #   
        df.append(airport_dic)
